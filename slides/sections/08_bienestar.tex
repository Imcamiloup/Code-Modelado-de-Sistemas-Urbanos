\begin{frame}{Validación y Análisis Dinámico}
\textbf{Script 1: \texttt{validacion\_sistema\_logistico.py}}:
\begin{itemize}
  \item Simula con $\theta^\star$ sobre años históricos
  \item Desnormaliza y calcula RMSE, MAE, $R^2$ por variable 
  \item Genera gráfico comparativo Observado vs Simulado
\end{itemize}

\textbf{Script 2: \texttt{analisis\_fase\_sistema\_logistico.py}}:
\begin{itemize}
  \item Diagramas de fase P-W con streamplot 
  \item Campos vectoriales muestran trayectorias cualitativas 
  \item Base para análisis de puntos fijos y estabilidad 
\end{itemize}

\textbf{Script 3: \texttt{proyeccion\_sistema\_logistico.py}}:
\begin{itemize}
  \item Proyección: 5 años con U, E constantes 
  \item Condición inicial: último valor histórico normalizado 
\end{itemize}
\end{frame}
